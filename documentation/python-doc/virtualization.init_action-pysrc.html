16  import sys 
17  sys.path.append("/usr/share/rhn/") 
18  from virtualization import support 
19   
20  actions =   { 
21                  'shutdown'  :   support.shutdown, 
22                  'start'     :   support.start, 
23                  'suspend'   :   support.suspend, 
24                  'resume'    :   support.resume, 
25                  'reboot'    :   support.reboot, 
26                  'destroy'   :   support.destroy, 
27              } 
28   
29  action_type = sys.argv[1] 
30  uuid = sys.argv[2] 
31   
32  if not action_type in list(actions.keys()): 
33      sys.stderr.write("Unknown action: %s \n" % action_type) 
34      sys.exit(1) 
35   
36  try: 
37      actions[action_type](uuid) 
38  except Exception: 
39      e = sys.exc_info()[1] 
40      sys.stderr.write(str(e)) 
41      sys.exit(1) 
42   
43  sys.exit(0)

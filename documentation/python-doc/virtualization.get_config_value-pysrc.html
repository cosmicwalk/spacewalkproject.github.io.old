16  import sys 
17  sys.path.append("/usr/share/rhn") 
18  from virtualization.domain_config import DomainConfig 
19   
20  fieldname = sys.argv[1] 
21  filename = sys.argv[2] 
22  dc = DomainConfig('/usr/share/rhn/virt/auto', filename) 
23   
24  fields = { 
25              'name'          :   DomainConfig.NAME, 
26              'uuid'          :   DomainConfig.UUID, 
27              'memory'        :   DomainConfig.MEMORY, 
28              'vcpu'          :   DomainConfig.VCPU, 
29              'root_device'   :   DomainConfig.ROOT_DEVICE, 
30              'cmdline'       :   DomainConfig.COMMAND_LINE, 
31              'os_kernel'     :   DomainConfig.KERNEL_PATH, 
32              'os_initrd'     :   DomainConfig.RAMDISK_PATH, 
33              'disk_source'   :   DomainConfig.DISK_IMAGE_PATH 
34           } 
35   
36  if fieldname not in fields: 
37      sys.stdout.write("Unknown configuration element %s \n" % fieldname) 
38      sys.exit(1) 
39   
40  result = dc.getConfigItem(fields[fieldname]) 
41  if fieldname == "uuid": 
42      result = result.replace("-", "") 
43   
44  print(result)

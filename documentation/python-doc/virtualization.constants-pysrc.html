51  ## 
52  # This structure maps the libvirt state enumeration to labels that 
53  # Red Hat Satellite understands. 
54  # Reasons we don't care about differences between NOSTATE, RUNNING and BLOCKED: 
55  # 1. technically, the domain is still "running" 
56  # 2. RHN Classic / Red Hat Satellite / Spacewalk are not able to 
57  # display 'blocked' & 'nostate' 
58  #    as valid states 
59  # 3. to avoid 'Abuse of Service' messages: bugs #230106 and #546676 
60   
61  VIRT_STATE_NAME_MAP = ( StateType.RUNNING,  # VIR_DOMAIN_NOSTATE 
62                          StateType.RUNNING,  # VIR_DOMAIN_RUNNING 
63                          StateType.RUNNING,  # VIR_DOMAIN_BLOCKED 
64                          StateType.PAUSED,   # VIR_DOMAIN_PAUSED 
65                          StateType.SHUTDOWN, # VIR_DOMAIN_SHUTDOWN 
66                          StateType.SHUTOFF,  # VIR_DOMAIN_SHUTOFF 
67                          StateType.CRASHED)  # VIR_DOMAIN_CRASHED

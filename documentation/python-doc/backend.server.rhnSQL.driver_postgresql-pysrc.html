 42      """ 
 43      Convert a query with named parameters (i.e. :id, :name, etc) into one 
 44      that uses %(id)s, %(name)s parameters instead. 
 45   
 46      python-psycopg2 requires parameters to be in this form, so to keep our 
 47      existing queries intact we'll convert them when provided to the

 50      RETURNS: the new query with parameters replaced 
 51      """ 
 52      log_debug(6, "Converting query for PostgreSQL: %s" % query)

 55      return new_query
